119template <typename T>
120class NeverDestroyed
121{
123 public: template <typename... Args>
124 explicit NeverDestroyed(Args &&... args)
125 {
126 // Uses "placement new" to construct a `T` in `storage_`.
127 new (&this->storage) T(std::forward<Args>(args)...);
128 }
129
131 public: ~NeverDestroyed() = default;
132
134 public: NeverDestroyed(const NeverDestroyed &) = delete;
135
137 public: NeverDestroyed(NeverDestroyed &&) = delete;
138
140 public: NeverDestroyed &operator=(const NeverDestroyed &) = delete;
141
143 public: NeverDestroyed &operator=(NeverDestroyed &&) noexcept = delete;
144
146 public: T &Access()
147 {
148 return *reinterpret_cast<T *>(&this->storage);
149 }
150
151 public: const T &Access() const
152 {
153 return *reinterpret_cast<const T *>(&this->storage);
154 }
155
156 private: typename std::aligned_storage<sizeof(T), alignof(T)>::type storage;
157};
